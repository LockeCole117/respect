def initialize(json_schema)
  @indent_level = 0
  @indent_size = 2
  @json_schema = json_schema
  @css_class ||= {
    json_highlight: "json_highlight",
    plain: "plain",
    key: "key",
    keyword: "keyword",
    string: "string",
    numeric: "numeric",
    comment: "comment",
  }
end

def dump(output = "")
  @output = output
  @output ||= String.new
  @output << "<div class=\"#{css_class[:json_highlight]}\"><pre>"
  @output << dump_json(@json_schema)
  @output << "</pre></div>\n"
  @output
end
